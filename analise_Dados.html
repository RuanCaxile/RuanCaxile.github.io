Visualizao de Dados

Por fim, a partir do conjunto de dados pode-se gerar Grficos que trazem Insights de forma visual.

In[10]:
"""
Esta clula  para o Grfico de Barras
categorizando por faixa de preo e vendo
a relao com a Quantidade vendida.
"""

# Definindo as faixas de preo
limites_de_faixas_de_preco = [39, 55, 71, 87, 103, 120]  # Definindo os limites das faixas de preo
faixas_de_preco = ['39-55', '55-71', '71-87', '87-103', '103-120']  # Definindo os rtulos das faixas de preo

# Criando uma nova coluna no DataFrame para as faixas de preo
data_frame['Faixa de Preo'] = pd.cut(data_frame['Preo Unitrio'], bins=limites_de_faixas_de_preco, labels=faixas_de_preco, include_lowest=True)

# Calculando a soma das quantidades vendidas por faixa de preo
faixa_preco_quantidade = data_frame.groupby('Faixa de Preo', observed=True)['Quantidade de Vendas'].sum().reset_index()



# Grfico de Barras para Quantidade de Vendas por Faixa de Preo
plt.figure(figsize=(10, 6))
ax = sns.barplot(x='Faixa de Preo',hue='Quantidade de Vendas', y='Quantidade de Vendas', data=faixa_preco_quantidade, palette='viridis', legend=False)

# Adicionando Rtulos de Dados para cada Barra presente no grfico de barras
for container in ax.containers:
    ax.bar_label(container)

# Configuraes do Grfico de Barras
plt.title('Quantidade de Vendas por Faixa de Preo')
plt.xlabel('Faixa de Preo')
plt.ylabel('Quantidade de Vendas')
plt.show()

"""
Esta clula  para o Grfico de Disperso
do Faturamento Total Dirio pelo Tempo
"""
# Converter o tipo de dado da coluna 'Data' para datetime
data_frame['Data'] = pd.to_datetime(data_frame['Data'])

# Calcular o faturamento para cada entrada
data_frame['Faturamento'] = data_frame['Quantidade de Vendas'] * data_frame['Preo Unitrio']

# Calcular o faturamento dirio total
faturamento_diario = data_frame.groupby('Data')['Faturamento'].sum().reset_index()

# Grfico de Disperso faturamento dirio total pelo tempo
plt.figure(figsize=(10, 6))
grafico_dispersao = sns.scatterplot(x='Data', y='Faturamento', data=faturamento_diario)
plt.title('Faturamento Dirio ao Longo do Tempo')
plt.xlabel('Data')
plt.ylabel('Faturamento')
plt.show()

"""
Esta clula ser utilizada 
para mostrar o Top 5 dos Cursos
que tem maior faturamento e tambm
quantos porcentos eles representam
do faturamento total, seguindo a lgica
da Regra de Pareto.
"""

# Cria o data frame para os 5 cursos com maiores faturamentos, agrupando pelo nome do curso e somando todos os faturamentos dirios para cada curso
top_5_cursos_com_maiores_faturamentos = data_frame.groupby('Nome do Curso')['Faturamento'].sum().reset_index().sort_values(by='Faturamento', ascending=False).head(5)

# Faturamento Total
faturamento_total = data_frame['Faturamento'].sum()

# Faturamento Total do Top 5 dos Cursos com Maiores Faturamentos
faturamento_total_top_5_cursos_com_maiores_faturamentos = top_5_cursos_com_maiores_faturamentos['Faturamento'].sum()

# Porcentagem que o Top 5 Cursos Com Maiores Faturamentos Representa do Faturamento Total
porcentagem_faturamento_top_5_cursos_com_maiores_faturamentos = 100*(faturamento_total_top_5_cursos_com_maiores_faturamentos/faturamento_total)

# Plotar o grfico de barras
plt.figure(figsize=(10, 6))
barras_top_5 = plt.bar(top_5_cursos_com_maiores_faturamentos['Nome do Curso'], top_5_cursos_com_maiores_faturamentos['Faturamento'], color='orange')

# Adicionar rtulos de dados (valores) acima das barras
for barra in barras_top_5:
    yval = barra.get_height()
    plt.text(barra.get_x() + barra.get_width()/2, yval, f'{yval:.2f}', ha='center', va='bottom', fontsize=9)


# Configuraes do Grfico
plt.xlabel('Cursos')
plt.ylabel('Faturamento')
plt.title('Top 5 Cursos por Faturamento')
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.show()

print(f"O Faturamento Total do Top 5 Cursos com Maiores Faturamentos representa {porcentagem_faturamento_top_5_cursos_com_maiores_faturamentos:.2f}% do Faturamento Total de R$ {faturamento_total:.2f}")

O Faturamento Total do Top 5 Cursos com Maiores Faturamentos representa 62.98% do Faturamento Total de R$ 32735.10
